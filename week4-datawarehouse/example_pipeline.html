üíª Sample Analytics Queries

-- Top 10 kategoriler (son 30 g√ºn)
SELECT
  c.category_name,
  COUNT(DISTINCT f.ad_key) as total_ads,
  SUM(f.view_count) as total_views,
  ROUND(AVG(f.price), 2) as avg_price
FROM fact_ad_activity f
JOIN dim_category c ON f.category_key = c.category_key
JOIN dim_date d ON f.date_key = d.date_key
WHERE d.date >= CURRENT_DATE - INTERVAL '30 days'
GROUP BY c.category_name
ORDER BY total_views DESC
LIMIT 10;
-- ≈ûehir bazƒ±nda ilan performansƒ±
WITH city_metrics AS (
  SELECT
    l.city,
    COUNT(*) as ad_count,
    SUM(f.view_count) as total_views,
    SUM(f.message_count) as total_messages,
    ROUND(AVG(f.price), 2) as avg_price
  FROM fact_ad_activity f
  JOIN dim_location l ON f.location_key = l.location_key
  GROUP BY l.city
)
SELECT
  city,
  ad_count,
  total_views,
  ROUND(total_messages::NUMERIC / NULLIF(total_views, 0) * 100, 2) as message_rate,
  avg_price
FROM city_metrics
ORDER BY ad_count DESC;
